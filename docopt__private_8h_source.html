168  virtual void fix_identities(UniquePatternSet& patterns) {
169  for(auto& child : fChildren) {
170  // this will fix up all its children, if needed
171  if (auto bp = dynamic_cast<BranchPattern*>(child.get())) {
172  bp->fix_identities(patterns);
173  }
174 
175  // then we try to add it to the list
176  auto inserted = patterns.insert(child);
177  if (!inserted.second) {
178  // already there? then reuse the existing shared_ptr for that thing
179  child = *inserted.first;
180  }
181  }
