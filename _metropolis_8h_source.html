256  auto CalculateA2(move_type move) const {
257  auto currentS3Action = S3_bulk_action(N1_TL_,
258  N3_31_,
259  N3_22_,
260  Alpha_,
261  K_,
262  Lambda_);
263  auto newS3Action = static_cast<Gmpzf>(0);
264  // auto newS3Action = static_cast<MP_Float>(0);
265  switch (move) {
266  case move_type::TWO_THREE:
267  // A (2,3) move removes a timelike edge and
268  // adds a (2,2) simplex

270  N3_31_,

272  Alpha_,
273  K_,
274  Lambda_);
275  break;
276  case move_type::THREE_TWO:
277  // A (3,2) move adds a timelike edge and
278  // removes a (2,2) simplex
279  newS3Action = S3_bulk_action(N1_TL_+1,
280  N3_31_,
281  N3_22_-1,
282  Alpha_,
283  K_,
284  Lambda_);
285  break;
286  case move_type::TWO_SIX:
287  // A (2,6) move adds 2 timelike edges and
288  // adds 2 (1,3) and 2 (3,1) simplices
289  newS3Action = S3_bulk_action(N1_TL_+2,
290  N3_31_+4,
291  N3_22_,
292  Alpha_,
293  K_,
294  Lambda_);
295  break;
296  case move_type::SIX_TWO:
297  // A (6,2) move removes 2 timelike edges and
298  // removes 2 (1,3) and 2 (3,1) simplices
299  newS3Action = S3_bulk_action(N1_TL_-2,
300  N3_31_,
301  N3_22_-4,
302  Alpha_,
303  K_,
304  Lambda_);
305  break;
306  case move_type::FOUR_FOUR:

308  return static_cast<Gmpzf>(1);
309  // return static_cast<MP_Float>(1);
310  }
