277  auto CalculateA2(move_type move) const noexcept {
278  auto currentS3Action = S3_bulk_action(N1_TL_,
279  N3_31_,
280  N3_22_,
281  Alpha_,
282  K_,
283  Lambda_);
284  auto newS3Action = static_cast<Gmpzf>(0);
285  // auto newS3Action = static_cast<MP_Float>(0);
286  switch (move) {
287  case move_type::TWO_THREE:
288  // A (2,3) move adds a timelike edge
289  // and a (2,2) simplex
290  newS3Action = S3_bulk_action(N1_TL_+1,
291  N3_31_,
292  N3_22_+1,
293  Alpha_,
294  K_,
295  Lambda_);
296  break;
297  case move_type::THREE_TWO:
298  // A (3,2) move removes a timelike edge
299  // and a (2,2) simplex
300  newS3Action = S3_bulk_action(N1_TL_-1,
301  N3_31_,
302  N3_22_-1,
303  Alpha_,
304  K_,
305  Lambda_);
306  break;
307  case move_type::TWO_SIX:
308  // A (2,6) move adds 2 timelike edges and
309  // 2 (1,3) and 2 (3,1) simplices
310  newS3Action = S3_bulk_action(N1_TL_+2,
311  N3_31_+4,
312  N3_22_,
313  Alpha_,
314  K_,
315  Lambda_);
316  break;
317  case move_type::SIX_TWO:
318  // A (6,2) move removes 2 timelike edges and
319  // 2 (1,3) and 2 (3,1) simplices
320  newS3Action = S3_bulk_action(N1_TL_-2,
321  N3_31_,
322  N3_22_-4,
323  Alpha_,
324  K_,
325  Lambda_);
326  break;
327  case move_type::FOUR_FOUR:
328  // A (4,4) move changes nothing with respect to the action,
329  // and e^0==1
330  #ifndef NDEBUG
331  std::cout << "A2 is 1" << std::endl;
332  #endif
333  return static_cast<Gmpzf>(1);
334  }
335 
336  auto exponent = newS3Action - currentS3Action;
337  auto exponent_double = Gmpzf_to_double(exponent);
338 
339  // if exponent > 0 then e^exponent >=1 so according to Metropolis
340  // algorithm return A2=1
341  if (exponent >=0) return static_cast<Gmpzf>(1);

361 
362  #ifndef NDEBUG

365
