162  move_name = "(2,6)";
163  break;
164  case move_type::SIX_TWO:

166  move_name = "(6,2)";
167  break;
168  }
169  // Set precision for initialization and assignment functions
170  mpfr_set_default_prec(PRECISION);
171 
172  // Initialize for MPFR
173  mpfr_t r1, r2, a1;
174  mpfr_inits2(PRECISION, r1, r2, a1, nullptr);
175 
176  mpfr_init_set_ui(r1, this_move, MPFR_RNDD); // r1 = this_move
177  mpfr_init_set_ui(r2, total_moves, MPFR_RNDD); // r2 = total_moves
178 
179  // The result
180  mpfr_div(a1, r1, r2, MPFR_RNDD); // a1 = r1/r2
