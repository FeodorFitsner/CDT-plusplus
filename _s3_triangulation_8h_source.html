458 inline void make_S3_triangulation(const unsigned number_of_simplices,
459  const unsigned number_of_timeslices,
460  const bool output,
461  Delaunay* const D3,

464  std::vector<Cell_handle>* const one_three)
465  noexcept {
466  std::cout << "Generating universe ..." << std::endl;
467  const auto simplices_per_timeslice = number_of_simplices /
468  number_of_timeslices;

470 
471  CGAL_triangulation_precondition(simplices_per_timeslice >= 1);
472 
473  const auto points = simplices_per_timeslice * 4;
474  const auto total_points = points * number_of_timeslices;
475  auto radius = 1.0;
476 
477  std::vector<Point> vertices;
478  std::vector<unsigned> timevalue;
479 
480  // We know how many points we have in advance, so reserve memory
481  vertices.reserve(total_points);
482  timevalue.reserve(total_points);
483 
484  for (auto i = 0; i < number_of_timeslices; ++i) {
485  radius = 1.0 + static_cast<double>(i);
486  // std::cout << "Loop: " << i << " Radius: " << radius << std::endl;
487  make_2_sphere(points, radius, output, &vertices, &timevalue);
488  }

492  // D3->insert(vertices.begin(), vertices.end());
493  // due to the timevalues
494  insert_into_S3(vertices, timevalue, D3);
495 
496  // Remove cells that have invalid foliations
497  auto pass = 0;
498  while (!check_timeslices(D3, output)) {
499  pass++;
500  if (pass > MAX_FOLIATION_FIX_PASSES) break;
501  std::cout << "Fix Pass #" << pass << std::endl;
502  fix_timeslices(output, D3);
503  }

513  << " cells." << std::endl;
514  std::cout << "There are " << three_one->size() << " (3,1) simplices" <<
515  " and " << two_two->size() << " (2,2) simplices and " <<
516  one_three->size() << " (1,3) simplices." << std::endl;
517  if (output) {
518  Delaunay::Finite_vertices_iterator vit;
519  for (vit = D3->finite_vertices_begin();

521  std::cout << "Point " << vit->point() << " has timeslice "

523  }
524  }
525  CGAL_triangulation_expensive_postcondition(D3->is_valid());
526 } // make_S3_triangulation()
