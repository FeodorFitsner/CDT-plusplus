163  for (cit = universe_ptr->finite_cells_begin();
164  cit != universe_ptr->finite_cells_end(); ++cit) {
165  std::list<unsigned> timevalues;

168  auto max_values = 0;
169  auto min_values = 0;
170  // Push every time value of every vertex into a list

172  timevalues.emplace_back(cit->vertex(i)->info());

346  causal_vertices.second.emplace_back(radius);
